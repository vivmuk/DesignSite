DEVELOPERS

Code that
shines.

First-class support for React, Vue, and Svelte. Drop in Neon components and watch them light up.

App.tsx
import { NeonCard, NeonButton, GlowText } from '@neon/ui';
import { useTheme } from '@neon/hooks';

export default function Dashboard() {
  const { glow } = useTheme();

  return (
    <NeonCard glow={"green"} animate>
      <GlowText as="h2" color={glow.primary>
        System Online
      </GlowText>
      <NeonButton variant="pulse">
        Activate
      </NeonButton>
    </NeonCard>
  );
}
